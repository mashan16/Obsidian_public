
#### Описание `sdXn`
X - это буква, которая уникально идентифицирует конкретное устройство:
n - это номер раздела на диске:
1, 2, 3, 4 и т.д. - номера логических разделов
Например:
sda1 - первый раздел на первом диске
sdb2 - второй раздел на втором диске
sdc5 - пятый раздел на третьем диске
#### Просмотр информации о разделах
   fdisk -l # показывает все разделы на всех дисках
   lsblk # показывает иерархию блочных устройств
   df -h # показывает использование файловых систем
   parted -l # показывает информацию о разделах
   blkid # показывает UUID и типы файловых систем
#### Создание/удаление разделов
  `fdisk /dev/sdXn` # интерактивное создание разделов
  `parted /dev/sdXn` # альтернативная утилита для создания разделов
  `mkfs.ext4 /dev/sdXn` # форматирование раздела в ext4
  `mkfs.xfs /dev/sdXn` # форматирование раздела в xfs
  `wipefs -a /dev/sdXn` # удаление меток файловой системы
#### Монтирование разделов
  `mount /dev/sdXn /mnt` # временно примонтировать раздел (чтение и запись)
  `mount -o ro,noload /dev/path-to /mnt/vm` # Временно примонтировать (только чтение)
  `umount /mnt` # отмонтировать раздел
  `mount -t ext4 /dev/sdXn /mnt` # монтирование с указанием типа ФС
  `mount --bind /source /destination` # монтирование одной директории в другую
##### Автомонтирование (fstab)
  nano /etc/fstab # редактирование fstab
  Пример строки:
  UUID=1234-5678 /mnt/ext hfsplus defaults 0 2
  # !!! Обязательно !!! следите за тем, чтобы в конце файла fstab был переход на новую пустую строку!! если не сделать система не загрузится!!!!

#### Тесты и проверки диска
  hdparm -tT /dev/sdX # Тестирует производительность чтения данных с диска /dev/sdX
  fsck /dev/sdX # Проверяет и исправляет ошибки в файловой системе на разделе /dev/sdX
  badblocks -sv /dev/sdX # Ищет BAD блоки на диске /dev/sdX
  dd if=/dev/sdX of=test bs=1M count=1024 # Для тестирования скорости записи (Создаёт файл test размером 1 ГБ)
  dd if=/dev/sdX of=/dev/sdX bs=1024k count=1024 # Тест скорости диска Вместо dev\sdc указывать нужный диск
#### Информация о дисках
  hdparm -I /dev/sdX # информация о диске
  smartctl -a /dev/sdX # SMART информация
  lsscsi # показывает SCSI устройства
#### Работа с LVM
  pvcreate /dev/sdXn # создание физического тома
  vgcreate myvg /dev/sdXn # создание группы томов
  lvcreate -L 10G myvg # создание логического тома
  lvextend -L +10G /dev/myvg/lv # расширение LV
  vgs # показывает группы томов
  pvs # показывает физические тома
  lvs # показывает логические тома
